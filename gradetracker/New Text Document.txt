<script>
// ── Data ──────────────────────────────────────────────────────────────────────
const DEFAULT_SUBJECTS = ['Mathematics','Science','English','History','Geography','Physics','Chemistry','Biology'];
const BAR_COLORS = ['#2a78d6','#1baf7a','#eda100','#4a3aa7','#e87ba4','#eb6834','#e34948','#1baf7a'];

let students = [];
let selectedId = null;

// ── Helpers ───────────────────────────────────────────────────────────────────
function avg(arr) { return arr.reduce((a, b) => a + b, 0) / arr.length; }
function maximum(arr) { return Math.max(...arr); }
function minimum(arr) { return Math.min(...arr); }

function letterGrade(a) {
  if (a >= 90) return 'A';
  if (a >= 80) return 'B';
  if (a >= 70) return 'C';
  if (a >= 60) return 'D';
  return 'F';
}

function gradeColor(g) {
  return { A: '#2a78d6', B: '#1baf7a', C: '#eda100', D: '#eb6834', F: '#e34948' }[g] || '#898781';
}

// ── Sample Data ───────────────────────────────────────────────────────────────
function loadSample() {
  const s = ['Mathematics', 'Science', 'English', 'History'];
  students = [
    { id: 'STU001', name: 'Alice Johnson',   subjects: s, scores: [92, 88, 95, 91] },
    { id: 'STU002', name: 'Bob Martinez',    subjects: s, scores: [75, 82, 68, 77] },
    { id: 'STU003', name: 'Carol White',     subjects: s, scores: [55, 60, 58, 62] },
    { id: 'STU004', name: 'David Lee',       subjects: s, scores: [98, 95, 97, 99] },
    { id: 'STU005', name: 'Emma Thompson',   subjects: s, scores: [83, 79, 86, 81] },
    { id: 'STU006', name: 'Frank Wilson',    subjects: s, scores: [45, 52, 48, 50] },
  ];
}

// ── Table ─────────────────────────────────────────────────────────────────────
function refreshTable() {
  const tbody = document.getElementById('tbody');
  const emptyState = document.getElementById('empty-state');
  tbody.innerHTML = '';

  if (!students.length) {
    emptyState.style.display = 'block';
    refreshStats();
    return;
  }
  emptyState.style.display = 'none';

  students.forEach(s => {
    const a = avg(s.scores);
    const h = maximum(s.scores);
    const l = minimum(s.scores);
    const g = letterGrade(a);
    const pass = a >= 60;
    const gc = gradeColor(g);

    const tr = document.createElement('tr');
    if (s.id === selectedId) tr.classList.add('selected');
    tr.setAttribute('tabindex', '0');
    tr.setAttribute('aria-label', s.name + ', average ' + a.toFixed(1) + '%, ' + (pass ? 'passing' : 'failing'));
    tr.onclick = () => selectStudent(s.id);
    tr.onkeydown = e => { if (e.key === 'Enter' || e.key === ' ') selectStudent(s.id); };

    tr.innerHTML = `
      <td style="color:#898781;font-size:12px">${escHtml(s.id)}</td>
      <td style="font-weight:500">${escHtml(s.name)}</td>
      <td style="color:#898781">${s.scores.length}</td>
      <td><span style="font-weight:500;color:#2a78d6">${a.toFixed(1)}%</span></td>
      <td style="color:#1baf7a">${h.toFixed(1)}%</td>
      <td style="color:#e34948">${l.toFixed(1)}%</td>
      <td><span class="grade-badge" style="background:${gc}20;color:${gc}">${g}</span></td>
      <td><span class="status-badge" style="background:${pass ? '#eaf3de' : '#fcebeb'};color:${pass ? '#3b6d11' : '#a32d2d'}">${pass ? 'Pass' : 'Fail'}</span></td>
    `;
    tbody.appendChild(tr);
  });
  refreshStats();
}

function selectStudent(id) {
  selectedId = (selectedId === id) ? null : id;
  refreshTable();
}

// ── Sidebar Stats ─────────────────────────────────────────────────────────────
function refreshStats() {
  const n = students.length;
  document.getElementById('sb-total').textContent = n;
  if (!n) {
    ['sb-avg', 'sb-top', 'sb-pass'].forEach(x => document.getElementById(x).textContent = '—');
    return;
  }
  const avgs = students.map(s => avg(s.scores));
  const classAvg = avg(avgs);
  const top = students[avgs.indexOf(Math.max(...avgs))];
  const passing = students.filter(s => avg(s.scores) >= 60).length;
  document.getElementById('sb-avg').textContent = classAvg.toFixed(1) + '%';
  document.getElementById('sb-top').textContent = top.name.split(' ')[0];
  document.getElementById('sb-pass').textContent = passing + ' / ' + n;
}

// ── Add Modal ─────────────────────────────────────────────────────────────────
function openAddModal() {
  document.getElementById('f-name').value = '';
  document.getElementById('f-id').value = '';
  document.getElementById('f-count').value = '4';
  buildSubjectFields();
  document.getElementById('add-modal').style.display = 'flex';
  setTimeout(() => document.getElementById('f-name').focus(), 50);
}

function closeAddModal() { document.getElementById('add-modal').style.display = 'none'; }
function closeAddIfBg(e) { if (e.target === document.getElementById('add-modal')) closeAddModal(); }

function buildSubjectFields() {
  const n = parseInt(document.getElementById('f-count').value);
  const container = document.getElementById('subject-fields');
  container.innerHTML = '';
  for (let i = 0; i < n; i++) {
    const def = DEFAULT_SUBJECTS[i] || ('Subject ' + (i + 1));
    const row = document.createElement('div');
    row.className = 'subjects-grid';
    row.innerHTML = `
      <input type="text" id="sn-${i}" value="${escHtml(def)}" placeholder="Subject name" aria-label="Subject ${i+1} name" />
      <input type="number" id="sc-${i}" placeholder="e.g. 85" min="0" max="100" step="0.1" aria-label="Subject ${i+1} score" />
    `;
    container.appendChild(row);
  }
}

function submitAdd() {
  const name = document.getElementById('f-name').value.trim();
  const id = document.getElementById('f-id').value.trim().toUpperCase();
  if (!name || !id) { showToast('Name and ID are required.'); return; }
  if (students.find(s => s.id === id)) { showToast('Student ID already exists.'); return; }

  const n = parseInt(document.getElementById('f-count').value);
  const scores = [], subjects = [];
  for (let i = 0; i < n; i++) {
    const sn = document.getElementById('sn-' + i).value.trim() || ('Subject ' + (i + 1));
    const sv = parseFloat(document.getElementById('sc-' + i).value);
    if (isNaN(sv) || sv < 0 || sv > 100) { showToast('Score for "' + sn + '" must be 0–100.'); return; }
    subjects.push(sn);
    scores.push(sv);
  }

  students.push({ id, name, subjects, scores });
  closeAddModal();
  selectedId = id;
  refreshTable();
  showToast(name + ' added to the roster.');
}

// ── Remove ────────────────────────────────────────────────────────────────────
function removeStudent() {
  if (!selectedId) { showToast('Select a student first.'); return; }
  const s = students.find(x => x.id === selectedId);
  if (!s) return;
  if (!confirm('Remove ' + s.name + ' from the roster?')) return;
  students = students.filter(x => x.id !== selectedId);
  selectedId = null;
  refreshTable();
  showToast('Student removed.');
}

// ── Report Modal ──────────────────────────────────────────────────────────────
function openReport() {
  if (!selectedId) { showToast('Select a student to view their report.'); return; }
  const s = students.find(x => x.id === selectedId);
  if (!s) return;

  const a = avg(s.scores);
  const h = maximum(s.scores);
  const l = minimum(s.scores);
  const g = letterGrade(a);
  const pass = a >= 60;

  // Bar chart
  const bars = s.scores.map((sc, i) => {
    const pct = (sc / 100) * 100;
    return `
      <div class="bar-wrap">
        <div class="bar-score">${sc.toFixed(0)}</div>
        <div class="bar-container">
          <div class="bar" style="height:${pct}%;background:${BAR_COLORS[i % BAR_COLORS.length]}"></div>
        </div>
        <div class="bar-label">${escHtml(s.subjects[i] || 'S' + (i+1))}</div>
      </div>`;
  }).join('');

  // Breakdown rows
  const rows = s.scores.map((sc, i) => {
    const lg = letterGrade(sc);
    const lc = gradeColor(lg);
    return `<tr>
      <td>${escHtml(s.subjects[i] || 'S' + (i+1))}</td>
      <td style="font-weight:500">${sc.toFixed(1)}</td>
      <td><span class="grade-badge" style="background:${lc}20;color:${lc}">${lg}</span></td>
    </tr>`;
  }).join('');

  document.getElementById('report-content').innerHTML = `
    <div class="report-header">
      <div class="report-name">${escHtml(s.name)}</div>
      <div class="report-meta">
        ID: ${escHtml(s.id)} &nbsp;·&nbsp; Letter grade: ${g}
        &nbsp;·&nbsp;
        <span style="background:${pass ? '#16a34a30' : '#dc262630'};padding:2px 10px;border-radius:20px;font-size:12px;font-weight:500">
          ${pass ? 'Pass' : 'Fail'}
        </span>
      </div>
    </div>
    <div class="stats-row">
      <div class="stat-card">
        <div class="stat-card-label">Average</div>
        <div class="stat-card-val" style="color:#2a78d6">${a.toFixed(2)}%</div>
      </div>
      <div class="stat-card">
        <div class="stat-card-label">Highest score</div>
        <div class="stat-card-val" style="color:#1baf7a">${h.toFixed(2)}%</div>
      </div>
      <div class="stat-card">
        <div class="stat-card-label">Lowest score</div>
        <div class="stat-card-val" style="color:#e34948">${l.toFixed(2)}%</div>
      </div>
    </div>
    <div class="chart-section">
      <div class="chart-title">Score by subject</div>
      <div class="bar-chart">${bars}</div>
    </div>
    <div class="breakdown-section">
      <div class="chart-title" style="margin-bottom:8px">Subject breakdown</div>
      <table aria-label="Subject scores for ${escHtml(s.name)}">
        <thead><tr><th>Subject</th><th>Score</th><th>Grade</th></tr></thead>
        <tbody>${rows}</tbody>
      </table>
    </div>
  `;
  document.getElementById('report-modal').style.display = 'flex';
  document.getElementById('report-modal').focus();
}

function closeReportModal() { document.getElementById('report-modal').style.display = 'none'; }
function closeReportIfBg(e) { if (e.target === document.getElementById('report-modal')) closeReportModal(); }

// ── Toast ─────────────────────────────────────────────────────────────────────
function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.style.display = 'block';
  clearTimeout(window._toastTimer);
  window._toastTimer = setTimeout(() => t.style.display = 'none', 2800);
}

// ── Escape HTML ───────────────────────────────────────────────────────────────
function escHtml(str) {
  return String(str).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

// ── Keyboard close modals ─────────────────────────────────────────────────────
document.addEventListener('keydown', e => {
  if (e.key === 'Escape') {
    if (document.getElementById('report-modal').style.display !== 'none') closeReportModal();
    else if (document.getElementById('add-modal').style.display !== 'none') closeAddModal();
  }
});
